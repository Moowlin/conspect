```python
def multiplication(a, b):
	"""
	(num, num) -> num 
	returns the product of the given numbers a and b 
	>>> multiplication(3, 5) 
	15 
	>>> multiplication(2, 6) 
	12
	"""
	s = a * b 
	return s
```
